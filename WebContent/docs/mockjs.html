拦截器(Mockjs)
该组件主要用于拦截AJAX请求，返回模拟数据。

用法介绍
layui.config({ 
  base: '/src/js/' 
}).use('mockjs',function(){
  var APIConfigs = {
    'GET /api/getuserinfo': {
      id: 1,
      name: '@name',
      email: '@email',
      address: '@region',
      datetime: '@datetime'
    },
    'POST /api/getlist': {
      'list|10': [{
        'id|+1': 1,
        name: '@name',
        email: '@email',
        datetime: '@datetime',
        address: '@region',
      }]
    },
  };
  // 注入mock
  layui.mockjs.inject(APIConfigs);
});
具体的配置方式请参考官方的文档： https://github.com/nuysoft/Mock/wiki/Getting-Started
